Public Function layer(nombre As Variant, Name As String, Color As Variant, LineType As String, LineWeight As Variant, Plottable As Boolean)
    Set nombre = AutoCAD.Application.ActiveDocument.Layers.Add(Name)
    nombre.Color = Color
    nombre.LineType = LineType
    nombre.LineWeight = LineWeight
    nombre.Plottable = Plottable
    nombre = nombre.Name
    'Color:
            'acByBlock
            'acRed
            'acYellow
            'acGreen
            'acCyan
            'acBlue
            'acMagenta
            'acWhite
    'LineType:
            'Continuous
            'ByLayer
            'ByBlock
    'LineWeight:
            'acLnWtByLayer
            'acLnWtByBlock
            'acLnWtByLwDefault
            'acLnWt000
            'acLnWt005
            'acLnWt009
            'acLnWt013
            'acLnWt015
            'acLnWt018
            'acLnWt020
            'acLnWt025
            'acLnWt030
            'acLnWt035
            'acLnWt040
            'acLnWt050
            'acLnWt053
            'acLnWt060
            'acLnWt070
            'acLnWt080
            'acLnWt090
            'acLnWt100
            'acLnWt106
            'acLnWt120
            'acLnWt140
            'acLnWt158
            'acLnWt200
            'acLnWt211
    
End Function
